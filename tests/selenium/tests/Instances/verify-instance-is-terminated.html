verify-instance-is-terminated
setSpeed 500
echo Verifies status is "Terminated" on instance detail page
assertVisible id=resource-menu-dashboard
clickAndWait id=resource-menu-dashboard
waitForVisible id=resource-menu-instances
clickAndWait id=resource-menu-instances
waitForVisible link=${instance_name} (${instance_id})
click link=${instance_name} (${instance_id})
storeEval 1 k
waitForVisible //form[@id='instance-form']/div/div[2]/span
storeText //form[@id='instance-form']/div/div[2]/span status
echo javascript{storedVars['status']=${status}};
echo ${status}
storeEval storedVars['terminated']="terminated";
echo ${terminated}
storeEval storedVars['status']==storedVars['terminated'] is_terminated
echo ${is_running}
echo javascript{storedVars['is_running']=${is_running}};
while ${k}<20&&!${is_terminated}
pause 20000
storeText //form[@id='instance-form']/div/div[2]/span status
echo ${status}
storeEval storedVars['status']==storedVars['terminated'] is_terminated
echo ${is_terminated}
storeEval ${k}+1 k
endWhile
echo ${instance_name} (${instance_id}) is running
