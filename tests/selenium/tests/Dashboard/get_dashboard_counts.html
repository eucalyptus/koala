get_dashboard_counts
setSpeed 500
echo Gets dashboard counts, including IAM users and groups for an admin user.
assertText link=Launch Instance Launch Instance
setTimeout 120000
waitForVisible id=resource-menu-dashboard
clickAndWait id=resource-menu-dashboard
waitForVisible id=running-instances-count
storeText id=running-instances-count running_instances
echo ${running_instances}
waitForVisible id=stopped-instances-count
storeText id=stopped-instances-count stopped_instances
echo ${stopped_instances}
waitForVisible id=securitygroups-count
storeText id=securitygroups-count s_groups
echo ${s_groups}
waitForVisible id=keypairs-count
storeText id=keypairs-count keypairs
echo ${keypairs}
waitForVisible id=stopped-instances-count
storeText id=stopped-instances-count stopped_instances
echo ${stopped_instances}
waitForVisible id=scalinggroup-instances-count
storeText id=scalinggroup-instances-count instances_in_scaling_groups
echo ${instances_in_scaling_groups}
waitForVisible id=ipaddresses-count
storeText id=ipaddresses-count ips
echo ${ips}
waitForVisible id=volumes-count
storeText id=volumes-count volumes
echo ${volumes}
waitForVisible id=snapshots-count
storeText id=snapshots-count snapshots
echo ${snapshots}
waitForVisible id=users-count
storeText id=users-count iam_users
echo ${iam_users}
waitForVisible id=groups-count
storeText id=groups-count iam_groups
echo ${iam_groups}
