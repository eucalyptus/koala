set-bfebs-image-for-lw
setSpeed 500
assertVisible id=resource-menu-dashboard
setTimeout 120000
echo Picks up an image with "centos" in name, if there is no such image looks for the one with "ubuntu" in name. Saves in "image variable the locator for "Select" button in launch wizard.
clickAndWait id=resource-menu-dashboard
waitForVisible id=resource-menu-autoscaling
click id=resource-menu-autoscaling
waitForVisible id=resource-menuitem-launchconfigs
clickAndWait id=resource-menuitem-launchconfigs
waitForVisible id=create-launchconfig-btn
clickAndWait id=create-launchconfig-btn
waitForVisible css=strong.image-title.ng-binding
storeElementPresent //td/div/strong[text()[contains(.,'bfebs')]] bfebs
echo ${bfebs}
gotoIf storedVars['bfebs'] == true END
gotoIf storedVars['bfebs'] == false NOIMAGE
label NOIMAGE
echo Could not find bfebs image
gotoIf true == true DONE
label END
store //td/div/strong[text()[contains(.,'bfebs')]] image
store ${image}/../../../td/a image
echo ${image}
label DONE
