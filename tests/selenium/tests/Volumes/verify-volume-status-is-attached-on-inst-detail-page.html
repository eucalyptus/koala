verify-volume-status-is-attached-on-inst-detail-page
setSpeed 500
assertVisible id=resource-menu-dashboard
setTimeout 300000
clickAndWait id=resource-menu-dashboard
waitForVisible //li[@id='instances-running']/div/div/a/i
clickAndWait //li[@id='instances-running']/div/div/a/i
waitForVisible //a[contains(text(),'${instance_id}')]
clickAndWait //a[contains(text(),'${instance_id}')]
waitForVisible //dl[@id='instance-subnav']/dd[2]/a
clickAndWait //dl[@id='instance-subnav']/dd[2]/a
storeEval 1 k
waitForVisible //a[@href="/volumes/${volume_id}"]/../../../div[3]
storeText //a[@href="/volumes/${volume_id}"]/../../../div[3] status
echo javascript{storedVars['status']=${status}};
echo ${status}
storeEval storedVars['attached']="attached";
echo ${attached}
storeEval storedVars['status']==storedVars['attached'] is_attached
echo ${is_attached}
echo javascript{storedVars['is_attached']=${is_attached}};
while ${k}<20&&!${is_attached}
pause 10000
storeText //a[@href="/volumes/${volume_id}"]/../../../div[3] status
echo ${status}
storeEval storedVars['status']==storedVars['attached'] is_attached
echo ${is_attached}
storeEval ${k}+1 k
endWhile
assertText //a[@href="/volumes/${volume_id}"]/../../../div[3] attached
